    public static void reverse (List<Integer> arr,int low,int high){
        int a = low;
        int b = high;
        while(a<=b){
            int temp = arr.get(a);
            arr.get(a) = arr.get(b);
            arr.get(b) = temp;
            a++;
            b--;
